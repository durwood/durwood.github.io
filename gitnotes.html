git tags -- List all tags
git checkout tags/<tag_name> -- checkout source code for given tag
git clean -fxd -- force clean of repository
git checkout master -- checkout latest code in master branch (undo detached head)
git checkout -- path/to/file -- restore file i.e. get rid of local changes
git checkout -b <MyBranch> -- Create local branch; switch to it; assoc current changes to it.
git push origin <MyBranch> -- Push code from local branch to remote, creating the remote branch.
git push --set-upstream origin <MyBranch> -- Push code from remote branch; set remote as upstream.
git merge <OtherBranch> -- Merge code from OtherBranch into current branch.
git branch [-r] --no-merged master -- List branches which have not been merged into master (-r for remoted branches).
git branch [-r] --merged master -- List branches which have been merged into master (-r for remoted branches).
